// 移动 合并相加
for(let s=0; s<a.length; s++>){
    i=0, j=0
    while(j<a[s].length>){// 实际就是4
        if(a[s][i] == -1){
            for(let k=j, k<4; k++){
                if(a[s][k] !== -1){
                    // 合并相加的逻辑， 每合并一次1；剩余格子数-1
                    if(i>0 && a[s][k] === a[s][i-1]){
                        a[s][i-1] += a[s][i-1]; // 合并相加
                        剩余格子数--；
                    } else {
                        a[s][i] = a[s][k];
                        i++;
                    }
                    a[s][k] = -1;
                    j = k+1;
                }
            }
        }else{
            i++;
        }
    }
}
